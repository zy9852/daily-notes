# 双重循环

function unique(arr) {
  let newArr = [arr[0]];
  for (let i = 1; i < arr.length; i++) {
    let flag = true;
    for (let j = 0; j < newArr.length; j++) {
      if (arr[i] === newArr[j]) {
        flag = false;
        break;
      }
    }
    if (flag) {
      newArr.push(arr[i]);
    }
  }
  return newArr;
}

# indexOf()

function unique(arr) {
  let newArr = [];
  for (let i = 0; i < arr.length; i++) {
    if (newArr.indexOf(arr[i]) === -1) {
      newArr.push(arr[i]);
    }
  }
  return newArr;
}
